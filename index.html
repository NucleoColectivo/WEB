  const carousel2 = document.getElementById('carousel-2');

  const visibles2 = 3;

  let currentIndex2 = 0;

  const total2 = carousel2.children.length;


  function moverCarrusel2(direccion) {

    currentIndex2 += direccion;

    if (currentIndex2 < 0) currentIndex2 = total2 - visibles2;

    if (currentIndex2 > total2 - visibles2) currentIndex2 = 0;

    const porcentaje = -(100 / visibles2) * currentIndex2;

    carousel2.style.transform = translateX(${porcentaje}%);

  }


  // Auto slide (opcional)

  setInterval(() => moverCarrusel2(1), 6000);


  // Soporte táctil (opcional)

  let startX2 = 0;

  document.getElementById("carousel-wrapper-2").addEventListener("touchstart", e => {

    startX2 = e.touches[0].clientX;

  });


  document.getElementById("carousel-wrapper-2").addEventListener("touchend", e => {

    const endX2 = e.changedTouches[0].clientX;

    if (startX2 - endX2 > 50) moverCarrusel2(1);

    else if (endX2 - startX2 > 50) moverCarrusel2(-1);

  });

</script>
